algoritmo "menu_opcoes"
var
   opcao: inteiro
inicio
   repita
      escreval("=== MENU ===")
      escreval("1 - Cadastrar")
      escreval("2 - Consultar")
      escreval("3 - Excluir")
      escreval("0 - Sair")
      escreva("Escolha uma opção: ")
      leia(opcao)

      escolha opcao
         caso 1
            escreval("Cadastrando...")
         caso 2
            escreval("Consultando...")
         caso 3
            escreval("Excluindo...")
         caso 0
            escreval("Saindo...")
         outrocaso
            escreval("Opção inválida!")
      fimescolha
      escreval()
   ate (opcao = 0)
fimalgoritmo
